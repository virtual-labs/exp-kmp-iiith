2.1 Class for Demo Instance

  • Creates a class which encapsulates all the variable which determine the state of the demo
  • We create an instance of the demo named DI (acronym for DemoInstance)
class demoInstance{
  constructor(){
    this.inString = null;
    this.lpsArray = null;
    this.i=null;
    this.j=null;
  };
};

let DI = new demoInstance() ;

2.2 Next

Called by next button and it moves the algorithm forward 

const next = function next(){
  resetColour();
  trackStringIndex(DI.i);
  if(DI.i<DI.inString.length){
    while(DI.j !== 0 && DI.inString[DI.i] !== DI.inString[DI.j]){
      DI.j = DI.lpsArray[DI.j-1];
    }

    if(DI.inString[DI.j] === DI.inString[DI.i]){
      DI.j++ ;
      DI.lpsArray[DI.i] = DI.j ;
      correctMatchRange(0,DI.lpsArray[DI.i]-1);
      correctMatchRange(DI.i-DI.lpsArray[DI.i]+1,DI.i);
    }
  }
  else {
    $("input[value='Next']").attr("disabled", true);
  }
  valueCardObjects[DI.i].innerHTML = DI.lpsArray[DI.i].toString() ;
  displayComment("The pattern scanned till here has a longest prefix which is also a suffix of length "+DI.lpsArray[DI.i] );
  DI.i++;
};

2.3 Setup

Called by the setup button. Uses the value of input fields to setup the demo and resets the state of the demo. 

const setup = function setup(){
  DI.inString = document.getElementById("patternInput").value;

  if(DI.inString.length > 15){
    displayComment("Input can't be longer than 15 characters long. Please try again.");
    return;
  }
  setupCards();
  $("input[value='Next']").attr("disabled", false);
  
  DI.i = 0;
  DI.j = 0;
  DI.lpsArray = Array(DI.inString.length).fill(0);
  trackStringIndex(0);
  DI.i = 1;
};

2.4 Standard implementation of the Preprocessing algorithm.

Not actually used in the program but for reference for the functioning of the code. 

const preProcessing = function preProcessing(inString){
  let j=0 ;
  let i=0 ;
  //A is the array which stores the longest prefix suffix
  let A = Array(inString.length).fill(0);
  
  for(i=1 ; i<inString.length ; i++){
    while(j!==0 && inString[i] !== inString[j]){
      j = A[j-1];
    }

    if(inString[j] === inString[i]){
      j++ ;
      A[i] = j ;
    }
  }
  return A;
};
